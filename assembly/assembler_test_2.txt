        lw      0       1       five        ; โหลดค่า five (5) จากหน่วยความจำ -> r1  (r1 = 5)
        lw      0       2       neg1        ; โหลดค่า -1 -> r2  (ไว้ใช้ลด r1)
        add     0       0       3           ; r3 = 0  (เริ่มผลรวม sum = 0)
loop    add     3       1       3           ; r3 = r3 + r1   (บวกค่าปัจจุบันของ r1 เข้าผลรวม)
        add     1       2       1           ; r1 = r1 + (-1)  => r1 = r1 - 1
        beq     0       1       done        ; ถ้า r1 == 0 ให้กระโดดไป done (เงื่อนไขจบลูป)
        beq     0       0       loop        ; เงื่อนไขจริงเสมอ (0 == 0) => วนกลับไป label loop
done    sw      0       3       result      ; เก็บผลรวม r3 ลงหน่วยความจำที่ label 'result'
        halt                                ; หยุดการทำงานของโปรแกรม
five    .fill   5                           ; ค่าคงที่ 5 (ตำแหน่งนี้ของหน่วยความจำถูกตั้งชื่อว่า five)
neg1    .fill   -1                          ; ค่าคงที่ -1
result  .fill   0                           ; ที่เก็บผลลัพธ์ เริ่มต้น 0 (จะถูกเขียนเป็น 15 หลังจบโปรแกรม)
